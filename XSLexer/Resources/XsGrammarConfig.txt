# XS Grammar rules
#
# Current Incoming Tokens:
# - Word	# Any word that isn't a reserved type
# - CbOp	# Curlybracket open {
# - CbCl	# Curlybracket close }
# - PaOp	# Parenthese open (
# - PaCl	# Parenthese close )
# - SqOp	# SquareBracket open [
# - SqCl	# SquareBracket close ]
# - Oper	# Operators
# - AsOp	# Assignment Operators
# - Asgn	# Assignment
# - Comp	# Comparison operatros
# - Cmma	# Comma
# - Numb	# Number
# - Coln	# Colon
# - Semc	# Semicolon
# - Stri	# String
# - Squo	# Single Quote
# - DatT	# DataType
# - KeyW	# Keyword

# ==========   Naming   ==========
# > Root
# = Start Definition
#
# ========== For Values ==========
# : followed by (not escapable char)
# ! Reference to this list
# & With value of
# * Multiple off (list)

> Func = DatT : Word : PaOp : !Param : PaCl
> Func = KeyW&mutable : DatT : Word : PaOp : !Param : PaCl
Statement = KeyW&if : PaOp : !Expr : PaCl : !Expr
Statement = KeyW&if : PaOP : !Expr : PaCl : CbOp : !*Expr : CbCl
Expr = !Value
Param = DatT: Word : Asng : !Value
Param = DatT: Word : Asng : !Value : Cmma : !Param
Value = Stri
Value = Numb
Value = Snqo
